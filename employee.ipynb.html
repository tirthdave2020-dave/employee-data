import sqlite3
import pandas as pd
conn=sqlite3.connect(':memory:')
query= """
CREATE TABLE employees (
    id INTEGER,
    name TEXT,
    department TEXT,
    salary INTEGER,
    experience_years INTEGER
    );
    """
conn.execute(query)
data=[
    (1, 'John', 'HR', 40000, 2),
    (2, 'sarah', 'sales', 55000, 4),
    (3, 'Mike', 'IT', 60000, 5),
    (4,'emma','sales',52000,3),
    (5,'david','IT',75000,7)
]
conn.executemany("INSERT INTO employees VALUES(?,?,?,?,?);",data)
conn.commit()
query="select * FROM employees;"
df=pd.read_sql_query(query,conn)
print(df)
df.fillna(0,inplace=True)
df['encoded_dept']=df['department'].map({'HR':1,'sales':2,'IT':3})
df['salary_per_year_exp']=df['salary']/df['experience_years']
import matplotlib.pyplot as plt
import seaborn as sns
plt.figure(figsize=(8,6))
sns.scatterplot(x='experience_years',y='salary',data=df,hue='department')
plt.title('Experience vs Salary')
plt.show()
sns.boxplot(x='department', y='salary',data=df)
plt.show()
